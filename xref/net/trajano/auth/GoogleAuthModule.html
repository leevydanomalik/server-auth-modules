3   import java.util.Map;
4   
5   import javax.json.Json;
6   import javax.security.auth.message.AuthException;
7   import javax.ws.rs.client.Client;
8   
9   import net.trajano.auth.internal.OpenIDProviderConfiguration;
10  
11  /**
12   * Google Server Auth Module. Uses Google services to do OAuth 2.0 Login. It
13   * extends {@link OAuthModule} to get the configuration from the module rather
14   * from Google.
15   *
16   * @author Archimedes Trajano
17   *
18   */
19  public class GoogleAuthModule extends OAuthModule {
20  
21      /**
22       * Build the OpenID config from META-INF/google/config.json.
23       *
24       * @param restClient
25       *            REST client
26       * @param options
27       *            ignored
28       * @return OpenID provider configuration
29       */
30      @Override
31      protected OpenIDProviderConfiguration getOpenIDProviderConfig(
32              final Client restClient, final Map<String, String> options)
33                      throws AuthException {
34          return new OpenIDProviderConfiguration(Json.createReader(
35                  Thread.currentThread().getContextClassLoader()
36                  .getResourceAsStream("META-INF/google-config.json"))
37                  .readObject());
38      }
39  }
