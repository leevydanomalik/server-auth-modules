23  /**
24   * Google Server Auth Module. Uses Google services to do OAuth 2.0 Login. It
25   * extends {@link OAuthModule} to get the configuration from the module rather
26   * from Google.
27   *
28   * @author Archimedes Trajano
29   *
30   */
31  public class GoogleAuthModule extends OAuthModule {
32  
33      /**
34       * Build the OpenID config from META-INF/google/config.json.
35       *
36       * @param restClient
37       *            REST client
38       * @param options
39       *            ignored
40       * @return OpenID provider configuration
41       */
42      @Override
43      protected OpenIDProviderConfiguration getOpenIDProviderConfig(
44              final Client restClient, final Map<String, String> options)
45                      throws AuthException {
46          return new OpenIDProviderConfiguration(Json.createReader(
47                  Thread.currentThread().getContextClassLoader()
48                  .getResourceAsStream("META-INF/google-config.json"))
49                  .readObject());
50      }
51  
52      /**
53       * Only does the passing of the authentication token via the POST parameters
54       * and does not try to do it via the authorization headers. {@inheritDoc}
55       */
56      @Override
57      protected OAuthToken getToken(final HttpServletRequest req,
58              final OpenIDProviderConfiguration oidProviderConfig)
59                      throws IOException {
60          final MultivaluedMap<String, String> requestData = new MultivaluedHashMap<>();
61          requestData.putSingle(CODE, req.getParameter("code"));
62          requestData.putSingle(GRANT_TYPE, "authorization_code");
63          requestData.putSingle(REDIRECT_URI, getRedirectionEndpointUri(req)
64                  .toASCIIString());
65          requestData.putSingle(CLIENT_ID, getClientId());
66          requestData.putSingle(CLIENT_SECRET_KEY, getClientSecret());
67          final OAuthToken authorizationTokenResponse = getRestClient()
68                  .target(oidProviderConfig.getTokenEndpoint())
69                  .request(MediaType.APPLICATION_JSON_TYPE)
70                  .post(Entity.form(requestData), OAuthToken.class);
71          return authorizationTokenResponse;
72      }
73  }
