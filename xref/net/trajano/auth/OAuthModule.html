258     /**
259      * Gets the ID token. This ensures that both cookies are present, if not
260      * then this will return <code>null</code>.
261      *
262      * @param req
263      *            HTTP servlet request
264      * @return ID token
265      * @throws GeneralSecurityException
266      * @throws IOException
267      */
268     private String getIdToken(final HttpServletRequest req) throws GeneralSecurityException, IOException {
269         final Cookie[] cookies = req.getCookies();
270         if (cookies == null) {
271             return null;
272         }
273         String idToken = null;
274         boolean foundAge = false;
275         for (final Cookie cookie : cookies) {
276             if (NET_TRAJANO_AUTH_ID.equals(cookie.getName()) && !isNullOrEmpty(cookie.getValue())) {
277                 idToken = cookie.getValue();
278             } else if (NET_TRAJANO_AUTH_AGE.equals(cookie.getName())) {
279                 if (!req.getRemoteAddr()
280                         .equals(new String(CipherUtil.decrypt(Base64.decode(cookie.getValue()), secret), "US-ASCII"))) {
281                     throw new AuthException(R.getString("ipaddressMismatch"));
282                 }
283                 foundAge = true;
284             }
285             if (idToken != null && foundAge) {
286                 return idToken;
287             }
288         }
289         return idToken;
290     }

305     /**
306      * This gets the redirection endpoint URI.
307      * <p>
308      * If the redirection endpoint URI option is not set then this gets the base
309      * URI for the application based on the request. This is used as the
310      * redirect URI for OAuth.
311      * <p>
312      * If the redirection endpoint is set, then it is resolved against the
313      * request URL.
314      *
315      * @param req
316      *            request
317      * @return redirection endpoint URI.
318      */
319     protected URI getRedirectionEndpointUri(final HttpServletRequest req) {
320         if (isNullOrEmpty(redirectionEndpointUri)) {
321             final StringBuffer redirectUri = req.getRequestURL();
322             // Get the third / character from the request URL should be the
323             // start of the path.
324             redirectUri.replace(redirectUri.indexOf("/", redirectUri.indexOf("/", redirectUri.indexOf("/") + 1) + 1), redirectUri.length(), req.getContextPath());
325             return URI.create(redirectUri.toString());
326         } else {
327             return URI.create(req.getRequestURL()
328                     .toString())
329                     .resolve(redirectionEndpointUri);
330         }
331     }

361     /**
362      * Sends a request to the token endpoint to get the token for the code.
363      *
364      * @param req
365      *            servlet request
366      * @param oidProviderConfig
367      *            OpenID provider config
368      * @return token response
369      */
370     protected OAuthToken getToken(final HttpServletRequest req, final OpenIDProviderConfiguration oidProviderConfig) throws IOException {
371         final MultivaluedMap<String, String> requestData = new MultivaluedHashMap<>();
372         requestData.putSingle(CODE, req.getParameter("code"));
373         requestData.putSingle(GRANT_TYPE, "authorization_code");
374         requestData.putSingle(REDIRECT_URI, getRedirectionEndpointUri(req).toASCIIString());
375 
376         try {
377             final String authorization = "Basic " + Base64.encode((clientId + ":" + clientSecret).getBytes("UTF8"));
378             final OAuthToken authorizationTokenResponse = restClient.target(oidProviderConfig.getTokenEndpoint())
379                     .request(MediaType.APPLICATION_JSON_TYPE)
380                     .header("Authorization", authorization)
381                     .post(Entity.form(requestData), OAuthToken.class);
382             if (LOG.isLoggable(Level.FINEST)) {
383                 LOG.finest("authorization token response =  " + authorizationTokenResponse);
384             }
385             return authorizationTokenResponse;
386         } catch (final BadRequestException e) {
387             // workaround for google that does not support BASIC authentication
388             // on their endpoint.
389             requestData.putSingle(CLIENT_ID, clientId);
390             requestData.putSingle(CLIENT_SECRET_KEY, clientSecret);
391             final OAuthToken authorizationTokenResponse = restClient.target(oidProviderConfig.getTokenEndpoint())
392                     .request(MediaType.APPLICATION_JSON_TYPE)
393                     .post(Entity.form(requestData), OAuthToken.class);
394             if (LOG.isLoggable(Level.FINEST)) {
395                 LOG.finest("authorization token response =  " + authorizationTokenResponse);
396             }
397             return authorizationTokenResponse;
398         }
399     }

419     /**
420      * Workaround for the issuer value for Google. This was documented in
421      * 15.6.2. of the spec. In which case if the issuer does not start with
422      * https:// it will prepend it.
423      *
424      * @param issuer
425      *            issuer
426      * @return updated issuer
427      */
428     private String googleWorkaround(final String issuer) {
429         if (issuer.startsWith(HTTPS_PREFIX)) {
430             return issuer;
431         }
432         return HTTPS_PREFIX + issuer;
433     }

435     /**
436      * Handles the callback.
437      *
438      * @param req
439      *            servlet request
440      * @param resp
441      *            servlet response
442      * @param subject
443      *            user subject
444      * @return status
445      * @throws GeneralSecurityException
446      */
447     private AuthStatus handleCallback(final HttpServletRequest req, final HttpServletResponse resp, final Subject subject) throws GeneralSecurityException, IOException {
448         final OpenIDProviderConfiguration oidProviderConfig = getOpenIDProviderConfig(restClient, moduleOptions);
449         final OAuthToken token = getToken(req, oidProviderConfig);
450         final JsonWebKeySet webKeys = getWebKeys(moduleOptions, oidProviderConfig);
451 
452         LOG.log(Level.FINEST, "tokenValue", token);
453         final JsonObject claimsSet = Json.createReader(new ByteArrayInputStream(Utils.getJwsPayload(token.getIdToken(), webKeys)))
454                 .readObject();
455 
456         validateIdToken(clientId, claimsSet);
457 
458         final String iss = googleWorkaround(claimsSet.getString("iss"));
459         final String issuer = googleWorkaround(oidProviderConfig.getIssuer());
460         if (!iss.equals(issuer)) {
461             LOG.log(Level.SEVERE, "issuerMismatch", new Object[] { iss, issuer });
462             throw new GeneralSecurityException(MessageFormat.format(R.getString("issuerMismatch"), iss, issuer));
463         }
464         updateSubjectPrincipal(subject, claimsSet);
465 
466         final TokenCookie tokenCookie;
467         if (oidProviderConfig.getUserinfoEndpoint() != null && Pattern.compile("\\bprofile\\b")
468                 .matcher(scope)
469                 .find()) {
470             final Response userInfoResponse = restClient.target(oidProviderConfig.getUserinfoEndpoint())
471                     .request(MediaType.APPLICATION_JSON_TYPE)
472                     .header("Authorization", token.getTokenType() + " " + token.getAccessToken())
473                     .get();
474             if (userInfoResponse.getStatus() == 200) {
475                 tokenCookie = new TokenCookie(claimsSet, userInfoResponse.readEntity(JsonObject.class));
476             } else {
477                 LOG.log(Level.WARNING, "unableToGetProfile");
478                 tokenCookie = new TokenCookie(claimsSet);
479             }
480         } else {
481             tokenCookie = new TokenCookie(claimsSet);
482         }
483 
484         final String requestCookieContext;
485         if (isNullOrEmpty(cookieContext)) {
486             requestCookieContext = req.getContextPath();
487         } else {
488             requestCookieContext = cookieContext;
489         }
490 
491         final Cookie idTokenCookie = new Cookie(NET_TRAJANO_AUTH_ID, tokenCookie.toCookieValue(clientId, clientSecret));
492         idTokenCookie.setMaxAge(-1);
493         idTokenCookie.setSecure(true);
494         idTokenCookie.setPath(requestCookieContext);
495         resp.addCookie(idTokenCookie);
496 
497         final SecretKey secret = CipherUtil.buildSecretKey(clientId, clientSecret);
498         final Cookie ageCookie = new Cookie(NET_TRAJANO_AUTH_AGE, Base64.encodeWithoutPadding(CipherUtil.encrypt(req.getRemoteAddr()
499                 .getBytes("US-ASCII"), secret)));
500         if (isNullOrEmpty(req.getParameter("expires_in"))) {
501             ageCookie.setMaxAge(3600);
502 
503         } else {
504             ageCookie.setMaxAge(Integer.parseInt(req.getParameter("expires_in")));
505         }
506         ageCookie.setPath(requestCookieContext);
507         idTokenCookie.setSecure(true);
508         resp.addCookie(ageCookie);
509 
510         final String stateEncoded = req.getParameter("state");
511         final String redirectUri = new String(Base64.decode(stateEncoded));
512         resp.sendRedirect(resp.encodeRedirectURL(redirectUri));
513 
514         return AuthStatus.SEND_SUCCESS;
515     }

517     /**
518      * {@inheritDoc}
519      *
520      * @param requestPolicy
521      *            request policy, ignored
522      * @param responsePolicy
523      *            response policy, ignored
524      * @param h
525      *            callback handler
526      * @param options
527      *            options
528      */
529     @SuppressWarnings("unchecked")
530     @Override
531     public void initialize(final MessagePolicy requestPolicy, final MessagePolicy responsePolicy, final CallbackHandler h, @SuppressWarnings("rawtypes") final Map options) throws AuthException {
532         handler = h;
533         try {
534             mandatory = requestPolicy.isMandatory();
535             moduleOptions = options;
536             clientId = moduleOptions.get(CLIENT_ID_KEY);
537             if (clientId == null) {
538                 LOG.log(Level.SEVERE, "missingOption", CLIENT_ID_KEY);
539                 throw new AuthException(MessageFormat.format(R.getString("missingOption"), CLIENT_ID_KEY));
540             }
541             cookieContext = moduleOptions.get(COOKIE_CONTEXT_KEY);
542             redirectionEndpointUri = moduleOptions.get(REDIRECTION_ENDPOINT_URI_KEY);
543             tokenUri = moduleOptions.get(TOKEN_URI_KEY);
544             userInfoUri = moduleOptions.get(USERINFO_URI_KEY);
545             scope = moduleOptions.get(SCOPE_KEY);
546             if (isNullOrEmpty(scope)) {
547                 scope = "openid";
548             }
549             clientSecret = moduleOptions.get(CLIENT_SECRET_KEY);
550             if (clientSecret == null) {
551                 LOG.log(Level.SEVERE, "missingOption", CLIENT_SECRET_KEY);
552                 throw new AuthException(MessageFormat.format(R.getString("missingOption"), CLIENT_SECRET_KEY));
553             }
554             LOGCONFIG.log(Level.CONFIG, "options", moduleOptions);
555             secret = CipherUtil.buildSecretKey(clientId, clientSecret);
556         } catch (final Exception e) {
557             // Should not happen
558             LOG.log(Level.SEVERE, "initializeException", e);
559             throw new AuthException(MessageFormat.format(R.getString("initializeException"), e.getMessage()));
560         }
561     }

563     /**
564      * Checks to see whether the {@link ServerAuthModule} is called by the
565      * resource owner. This is indicated by the presence of a <code>code</code>
566      * and a <code>state</code> on the URL and is an idempotent request (i.e.
567      * GET or HEAD). The resource owner would be a web browser that got a
568      * redirect sent by the OAuth 2.0 provider.
569      *
570      * @param req
571      *            HTTP servlet request
572      * @return the module is called by the resource owner.
573      */
574     public boolean isCallback(final HttpServletRequest req) {
575         return moduleOptions.get(REDIRECTION_ENDPOINT_URI_KEY)
576                 .equals(req.getRequestURI()) && isRetrievalRequest(req) && !isNullOrEmpty(req.getParameter(CODE)) && !isNullOrEmpty(req.getParameter(STATE));
577     }

579     /**
580      * Builds the token cookie and updates the subject principal and sets the
581      * user info attribute in the request. Any exceptions or validation problems
582      * during validation will make this return <code>null</code> to indicate
583      * that there was no valid token.
584      *
585      * @param subject
586      *            subject
587      * @param req
588      *            servlet request
589      * @return token cookie.
590      */
591     private TokenCookie processTokenCookie(final Subject subject, final HttpServletRequest req) {
592         try {
593             final String idToken = getIdToken(req);
594             TokenCookie tokenCookie = null;
595             if (idToken != null) {
596                 tokenCookie = new TokenCookie(idToken, clientId, clientSecret);
597                 validateIdToken(clientId, tokenCookie.getIdToken());
598                 updateSubjectPrincipal(subject, tokenCookie.getIdToken());
599 
600                 if (tokenCookie.getUserInfo() != null) {
601                     req.setAttribute(NET_TRAJANO_AUTH_USERINFO, tokenCookie.getUserInfo());
602                 }
603 
604             }
605             return tokenCookie;
606         } catch (final GeneralSecurityException | IOException e) {
607             LOG.log(Level.FINE, "invalidToken", e.getMessage());
608             LOG.throwing(this.getClass()
609                     .getName(), "validateRequest", e);
610             return null;
611         }
612     }

614     /**
615      * Sends a redirect to the authorization endpoint. It sends the current
616      * request URI as the state so that the user can be redirected back to the
617      * last place. However, this does not work for non-idempotent requests such
618      * as POST in those cases it will result in a 401 error and
619      * {@link AuthStatus#SEND_FAILURE}. For idempotent requests, it will build
620      * the redirect URI and return {@link AuthStatus#SEND_CONTINUE}.
621      *
622      * @param req
623      *            HTTP servlet request
624      * @param resp
625      *            HTTP servlet response
626      * @return authentication status
627      * @throws AuthException
628      */
629     private AuthStatus redirectToAuthorizationEndpoint(final HttpServletRequest req, final HttpServletResponse resp) throws AuthException {
630         URI authorizationEndpointUri = null;
631         try {
632             final OpenIDProviderConfiguration oidProviderConfig = getOpenIDProviderConfig(restClient, moduleOptions);
633             restClient.close();
634 
635             final String state = Base64.encodeWithoutPadding(req.getRequestURI()
636                     .getBytes("UTF-8"));
637 
638             authorizationEndpointUri = UriBuilder.fromUri(oidProviderConfig.getAuthorizationEndpoint())
639                     .queryParam(CLIENT_ID, clientId)
640                     .queryParam(RESPONSE_TYPE, "code")
641                     .queryParam(SCOPE, scope)
642                     .queryParam(REDIRECT_URI, URI.create(req.getRequestURL()
643                             .toString())
644                             .resolve(moduleOptions.get(REDIRECTION_ENDPOINT_URI_KEY)))
645                     .queryParam(STATE, state)
646                     .build();
647 
648             resp.sendRedirect(authorizationEndpointUri.toASCIIString());
649             return AuthStatus.SEND_CONTINUE;
650         } catch (final IOException e) {
651             // Should not happen
652             LOG.log(Level.SEVERE, "sendRedirectException", new Object[] { authorizationEndpointUri, e.getMessage() });
653             LOG.throwing(this.getClass()
654                     .getName(), "redirectToAuthorizationEndpoint", e);
655             throw new AuthException(MessageFormat.format(R.getString("sendRedirectException"), authorizationEndpointUri, e.getMessage()));
656         }
657     }

685     /**
686      * Updates the principal for the subject. This is done through the
687      * callbacks.
688      *
689      * @param subject
690      *            subject
691      * @param jwtPayload
692      *            JWT payload
693      * @throws AuthException
694      * @throws GeneralSecurityException
695      */
696     private void updateSubjectPrincipal(final Subject subject, final JsonObject jwtPayload) throws GeneralSecurityException {
697         try {
698             final String iss = googleWorkaround(jwtPayload.getString("iss"));
699             handler.handle(new Callback[] { new CallerPrincipalCallback(subject, UriBuilder.fromUri(iss)
700                     .userInfo(jwtPayload.getString("sub"))
701                     .build()
702                     .toASCIIString()), new GroupPrincipalCallback(subject, new String[] { iss }) });
703         } catch (final IOException | UnsupportedCallbackException e) {
704             // Should not happen
705             LOG.log(Level.SEVERE, "updatePrincipalException", e.getMessage());
706             LOG.throwing(this.getClass()
707                     .getName(), "updateSubjectPrincipal", e);
708             throw new AuthException(MessageFormat.format(R.getString("updatePrincipalException"), e.getMessage()));
709         }
710     }

712     /**
713      * Validates the request. The request must be secure otherwise it will
714      * return {@link AuthStatus#FAILURE}. It then tries to build the token
715      * cookie data if available, if the token is valid, subject is set correctly
716      * and user info data if present is stored in the request, then call HTTP
717      * method specific operations.
718      *
719      * @param messageInfo
720      *            request and response
721      * @param clientSubject
722      *            client subject
723      * @param serviceSubject
724      *            service subject, ignored.
725      * @return Auth status
726      */
727     @Override
728     public AuthStatus validateRequest(final MessageInfo messageInfo, final Subject clientSubject, final Subject serviceSubject) throws AuthException {
729         final HttpServletRequest req = (HttpServletRequest) messageInfo.getRequestMessage();
730 
731         final HttpServletResponse resp = (HttpServletResponse) messageInfo.getResponseMessage();
732 
733         try {
734             final TokenCookie tokenCookie = processTokenCookie(clientSubject, req);
735 
736             if (tokenCookie != null && req.isSecure() && isGetRequest(req) && req.getRequestURI()
737                     .equals(tokenUri)) {
738                 resp.setContentType(MediaType.APPLICATION_JSON);
739                 resp.getWriter()
740                         .print(tokenCookie.getIdToken());
741                 return AuthStatus.SEND_SUCCESS;
742             }
743 
744             if (tokenCookie != null && req.isSecure() && isGetRequest(req) && req.getRequestURI()
745                     .equals(userInfoUri)) {
746                 resp.setContentType(MediaType.APPLICATION_JSON);
747                 resp.getWriter()
748                         .print(tokenCookie.getUserInfo());
749                 return AuthStatus.SEND_SUCCESS;
750             }
751 
752             if (!mandatory && !req.isSecure()) {
753                 // successful if the module is not mandatory and the channel is
754                 // not secure.
755                 return AuthStatus.SUCCESS;
756             }
757 
758             if (!req.isSecure() && mandatory) {
759                 // Fail authorization 3.1.2.1
760                 resp.sendError(HttpURLConnection.HTTP_FORBIDDEN, R.getString("SSLReq"));
761                 return AuthStatus.SEND_FAILURE;
762             }
763 
764             //
765 
766             if (!req.isSecure() && isCallback(req)) {
767                 resp.sendError(HttpURLConnection.HTTP_FORBIDDEN, R.getString("SSLReq"));
768                 return AuthStatus.SEND_FAILURE;
769             }
770 
771             if (req.isSecure() && isCallback(req)) {
772                 return handleCallback(req, resp, clientSubject);
773             }
774 
775             if (!mandatory || tokenCookie != null && !tokenCookie.isExpired()) {
776                 return AuthStatus.SUCCESS;
777             }
778             if (req.isSecure() && isHeadRequest(req) && req.getRequestURI()
779                     .equals(tokenUri)) {
780                 resp.setContentType(MediaType.APPLICATION_JSON);
781                 return AuthStatus.SEND_SUCCESS;
782             }
783 
784             if (req.getRequestURI()
785                     .equals(userInfoUri) && isHeadRequest(req)) {
786                 resp.setContentType(MediaType.APPLICATION_JSON);
787                 return AuthStatus.SEND_SUCCESS;
788 
789             }
790             if (!isRetrievalRequest(req)) {
791                 resp.sendError(HttpURLConnection.HTTP_FORBIDDEN, "Unable to POST when unauthorized.");
792                 return AuthStatus.SEND_FAILURE;
793             }
794 
795             return redirectToAuthorizationEndpoint(req, resp);
796         } catch (final Exception e) {
797             e.printStackTrace();
798             // Should not happen
799             LOG.log(Level.SEVERE, "validationException", e.getMessage());
800             LOG.throwing(this.getClass()
801                     .getName(), "validateRequest", e);
802             throw new AuthException(MessageFormat.format(R.getString("validationException"), e.getMessage()));
803         }
804     }
