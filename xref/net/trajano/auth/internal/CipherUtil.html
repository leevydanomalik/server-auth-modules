3   import java.io.ByteArrayInputStream;
4   import java.io.ByteArrayOutputStream;
5   import java.io.IOException;
6   import java.io.InputStream;
7   import java.io.OutputStream;
8   import java.security.GeneralSecurityException;
9   import java.util.zip.GZIPInputStream;
10  import java.util.zip.GZIPOutputStream;
11  
12  import javax.crypto.Cipher;
13  import javax.crypto.CipherInputStream;
14  import javax.crypto.CipherOutputStream;
15  import javax.crypto.SecretKey;
16  import javax.crypto.SecretKeyFactory;
17  import javax.crypto.spec.PBEKeySpec;
18  import javax.crypto.spec.SecretKeySpec;
19  
20  /**
21   * Utility class to decrypt and encrypt data. It is a compressing stream.
22   */
23  public final class CipherUtil {
24      /**
25       * Cipher algorithm to use. "AES"
26       */
27      private static final String CIPHER_ALGORITHM = "AES";
28  
29      public static InputStream buildDecryptStream(final InputStream inputStream, final SecretKey secret) throws GeneralSecurityException, IOException {
30          final Cipher cipher = Cipher.getInstance(CIPHER_ALGORITHM);
31          cipher.init(Cipher.DECRYPT_MODE, secret);
32          return new GZIPInputStream(new CipherInputStream(inputStream, cipher));
33      }
34  
35      public static OutputStream buildEncryptStream(final OutputStream outputStream, final SecretKey secret) throws GeneralSecurityException, IOException {
36          final Cipher cipher = Cipher.getInstance(CIPHER_ALGORITHM);
37          cipher.init(Cipher.ENCRYPT_MODE, secret);
38          return new GZIPOutputStream(new CipherOutputStream(outputStream, cipher));
39      }
40  
41      /**
42       * Build secret key.
43       *
44       * @param clientId
45       *            client ID (used for creating the {@link SecretKey})
46       * @param clientSecret
47       *            client secret (used for creating the {@link SecretKey})
48       * @return a secret key
49       * @throws GeneralSecurityException
50       *             crypto API problem
51       */
52      public static SecretKey buildSecretKey(final String clientId, final String clientSecret) throws GeneralSecurityException {
53          final PBEKeySpec pbeSpec = new PBEKeySpec(clientSecret.toCharArray(), clientId.getBytes(), 42, 128);
54  
55          final SecretKeyFactory factory = SecretKeyFactory.getInstance("PBKDF2WithHmacSHA1");
56          return new SecretKeySpec(factory.generateSecret(pbeSpec)
57                  .getEncoded(), CIPHER_ALGORITHM);
58      }
59  
60      public static byte[] decrypt(final byte[] cipherText, final SecretKey secret) throws GeneralSecurityException, IOException {
61          final ByteArrayInputStream is = new ByteArrayInputStream(cipherText);
62          final InputStream zis = CipherUtil.buildDecryptStream(is, secret);
63  
64          final ByteArrayOutputStream baos = new ByteArrayOutputStream(2000);
65  
66          int ch = zis.read();
67          while (ch != -1) {
68              baos.write(ch);
69              ch = zis.read();
70          }
71          zis.close();
72          baos.close();
73          return baos.toByteArray();
74      }
75  
76      public static byte[] encrypt(final byte[] clearText, final SecretKey secret) throws GeneralSecurityException, IOException {
77          final ByteArrayOutputStream baos = new ByteArrayOutputStream(2000);
78          final OutputStream zos = CipherUtil.buildEncryptStream(baos, secret);
79          zos.write(clearText);
80          zos.close();
81          return baos.toByteArray();
82      }
83  
84      /**
85       * Prevent instantiation of utility class.
86       */
87      private CipherUtil() {
88      }
89  
90  }
