36      /**
37       * Gets the JWS Payload from a <a href=
38       * "http://tools.ietf.org/html/draft-ietf-jose-json-web-signature-30#section-3.1"
39       * >JWS Compact Serialization</a>. The validation follows the rules in <a
40       * href=
41       * "http://tools.ietf.org/html/draft-ietf-jose-json-web-signature-30#section-5.2"
42       * >Message Signature or MAC validation section of JSON Web Signature</a>.
43       * <p>
44       * Note that "jku", "jwk", "x5u" and "x5c" should nor will <b>never</b> be
45       * implemented. It does not make sense for the serialization to contain its
46       * own validation.
47       *
48       * @param serialization
49       *            JWS compact serialization
50       * @param keyset
51       *            JSON web key used to validate the token
52       * @return the JWS payload
53       * @throws GeneralSecurityException
54       *             problem with crypto APIs or signature was not valid
55       */
56      public static byte[] getJwsPayload(final String serialization, final JsonWebKeySet keyset) throws GeneralSecurityException {
57          if (LOG.isLoggable(Level.FINEST)) {
58              LOG.finest("serialized payload = " + serialization);
59          }
60          final String[] jwtParts = serialization.split("\\.");
61  
62          final JsonObject joseHeader = Json.createReader(new ByteArrayInputStream(Base64.decode(jwtParts[0])))
63                  .readObject();
64  
65          // Handle plaintext JWTs
66          if (!"none".equals(joseHeader.getString("alg"))) {
67  
68              final String kid;
69              if (joseHeader.containsKey("kid")) {
70                  kid = joseHeader.getString("kid");
71              } else {
72                  kid = "";
73              }
74              final PublicKey signingKey = keyset.getKey(kid, PublicKey.class);
75  
76              if (signingKey == null) {
77                  throw new GeneralSecurityException("No key with id " + kid + " defined");
78              }
79  
80              final Signature signature = Signature.getInstance(toJavaAlgorithm(joseHeader.getString("alg")));
81  
82              final byte[] jwtSignatureBytes = Base64.decode(jwtParts[2]);
83  
84              signature.initVerify(signingKey);
85              signature.update((jwtParts[0] + "." + jwtParts[1]).getBytes());
86              if (!signature.verify(jwtSignatureBytes)) {
87                  throw new GeneralSecurityException("signature verification failed");
88              }
89          }
90          return Base64.decode(jwtParts[1]);
91      }

93      /**
94       * Checks if the request uses the GET method.
95       *
96       * @param req
97       *            request
98       * @return <code>true</code> if the request uses the GET method.
99       */
100     public static boolean isGetRequest(final HttpServletRequest req) {
101         return "GET".equals(req.getMethod());
102     }
103 
104     /**
105      * Checks if the request uses the HEAD method.
106      *
107      * @param req
108      *            request
109      * @return <code>true</code> if the request uses the HEAD method.
110      */
111     public static boolean isHeadRequest(final HttpServletRequest req) {
112         return "HEAD".equals(req.getMethod());
113     }

115     /**
116      * Checks if string is null or empty.
117      *
118      * @param s
119      *            string to test
120      * @return true if string is null or empty.
121      */
122     public static boolean isNullOrEmpty(final String s) {
123         return s == null || s.trim()
124                 .length() == 0;
125     }
126 
127     /**
128      * Checks if the request is to retrieve data (i.e. "GET" or "HEAD" method).
129      *
130      * @param req
131      *            request
132      * @return <code>true</code> if the request uses the GET or HEAD method.
133      */
134     public static boolean isRetrievalRequest(final HttpServletRequest req) {
135         return isGetRequest(req) || isHeadRequest(req);
136     }

138     /**
139      * Maps <a href=
140      * "http://tools.ietf.org/html/draft-ietf-jose-json-web-algorithms-29">JSON
141      * Web Algorithm</a> names to <a href=
142      * "http://docs.oracle.com/javase/7/docs/technotes/guides/security/SunProviders.html"
143      * >Java Crypto</a> names. If the mapping is not known then the original
144      * value is returned.
145      *
146      * @param alg
147      *            algorithm name
148      * @return algorithm name
149      */
150     public static String toJavaAlgorithm(final String alg) {
151         return JsonWebAlgorithm.valueOf(alg)
152                 .toJca();
153     }

155     /**
156      * Validates the ID Token.
157      *
158      * @param clientId
159      *            client ID
160      * @param idTokenJson
161      *            ID Token JSON.
162      * @throws GeneralSecurityException
163      */
164     public static void validateIdToken(final String clientId, final JsonObject idTokenJson) throws GeneralSecurityException {
165         // TODO handle multiple audiences
166         if (!clientId.equals(idTokenJson.getString("aud"))) {
167             throw new GeneralSecurityException(String.format("invalid 'aud' got' %s' expected '%s'", idTokenJson.getString("aud"), clientId));
168         }
169         if (idTokenJson.containsKey("azp") && !clientId.equals(idTokenJson.getString("azp"))) {
170             throw new GeneralSecurityException(String.format("invalid 'azp' got' %s' expected '%s'", idTokenJson.getString("azp"), clientId));
171         }
172         if (idTokenJson.containsKey("exp")) {
173             final long delta = System.currentTimeMillis() - idTokenJson.getInt("exp") * 1000L;
174             if (delta >= 0) {
175                 throw new GeneralSecurityException("expired " + delta + "ms ago");
176             }
177         }
178     }
