20      /**
21       * Gets the JWS Payload from a <a href=
22       * "http://tools.ietf.org/html/draft-ietf-jose-json-web-signature-30#section-3.1"
23       * >JWS Compact Serialization</a>. The validation follows the rules in <a
24       * href=
25       * "http://tools.ietf.org/html/draft-ietf-jose-json-web-signature-30#section-5.2"
26       * >Message Signature or MAC validation section of JSON Web Signature</a>.
27       * <p>
28       * Note that "jku", "jwk", "x5u" and "x5c" should nor will <b>never</b> be
29       * implemented. It does not make sense for the serialization to contain its
30       * own validation.
31       *
32       * @param serialization
33       *            JWS compact serialization
34       * @param keyset
35       *            JSON web key used to validate the token
36       * @return the JWS payload
37       * @throws GeneralSecurityException
38       *             problem with crypto APIs or signature was not valid
39       */
40      public static byte[] getJwsPayload(final String serialization,
41              final JsonWebKeySet keyset) throws GeneralSecurityException {
42          final String[] jwtParts = serialization.split("\\.");
43  
44          final JsonObject joseHeader = Json.createReader(
45                  new ByteArrayInputStream(Base64.decode(jwtParts[0])))
46                  .readObject();
47  
48          // Handle plaintext JWTs
49          if (!"none".equals(joseHeader.getString("alg"))) {
50  
51              final String kid;
52              if (joseHeader.containsKey("kid")) {
53                  kid = joseHeader.getString("kid");
54              } else {
55                  kid = "";
56              }
57              final PublicKey signingKey = keyset.getKey(kid, PublicKey.class);
58  
59              if (signingKey == null) {
60                  throw new GeneralSecurityException("No key with id " + kid
61                          + " defined");
62              }
63  
64              final Signature signature = Signature
65                      .getInstance(toJavaAlgorithm(joseHeader.getString("alg")));
66  
67              final byte[] jwtSignatureBytes = Base64.decode(jwtParts[2]);
68  
69              signature.initVerify(signingKey);
70              signature.update((jwtParts[0] + "." + jwtParts[1]).getBytes());
71              if (!signature.verify(jwtSignatureBytes)) {
72                  throw new GeneralSecurityException(
73                          "signature verification failed");
74              }
75          }
76          return Base64.decode(jwtParts[1]);
77      }

79      /**
80       * Checks if string is null or empty.
81       *
82       * @param s
83       *            string to test
84       * @return true if string is null or empty.
85       */
86      public static boolean isNullOrEmpty(final String s) {
87          return s == null || s.trim().length() == 0;
88      }

90      /**
91       * Maps <a href=
92       * "http://tools.ietf.org/html/draft-ietf-jose-json-web-algorithms-29">JSON
93       * Web Algorithm</a> names to <a href=
94       * "http://docs.oracle.com/javase/7/docs/technotes/guides/security/SunProviders.html"
95       * >Java Crypto</a> names. If the mapping is not known then the original
96       * value is returned.
97       *
98       * @param alg
99       *            algorithm name
100      * @return algorithm name
101      */
102     public static String toJavaAlgorithm(final String alg) {
103         if ("RS256".equalsIgnoreCase(alg)) {
104             return "SHA256withRSA";
105         }
106         return alg;
107     }

109     /**
110      * Validates the ID Token.
111      *
112      * @param clientId
113      *            client ID
114      * @param idTokenJson
115      *            ID Token JSON.
116      * @throws GeneralSecurityException
117      */
118     public static void validateIdToken(final String clientId,
119             final JsonObject idTokenJson) throws GeneralSecurityException {
120         // TODO handle multiple audiences
121         if (!clientId.equals(idTokenJson.getString("aud"))) {
122             throw new GeneralSecurityException(String.format(
123                     "invalid 'aud' got' %s' expected '%s'",
124                     idTokenJson.getString("aud"), clientId));
125         }
126         if (idTokenJson.containsKey("azp")
127                 && !clientId.equals(idTokenJson.getString("azp"))) {
128             throw new GeneralSecurityException(String.format(
129                     "invalid 'azp' got' %s' expected '%s'",
130                     idTokenJson.getString("azp"), clientId));
131         }
132         if (idTokenJson.containsKey("exp")
133                 && System.currentTimeMillis() > idTokenJson.getInt("exp") * 1000L) {
134             throw new GeneralSecurityException("expired");
135         }
136     }
