package net.trajano.auth.filter;

import java.util.HashMap;
import java.util.Map;

import javax.security.auth.message.MessageInfo;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

/**
 * MessageInfo that contains HttpServletRequest and HttpServletResponse.
 */
public class HttpServletRequestResponseMessageInfo implements MessageInfo {
    /**
     * Map.
     */
    private final Map<?, ?> map = new HashMap<>();
    /**
     * Request.
     */
    private HttpServletRequest request;
    /**
     * Response.
     */
    private HttpServletResponse response;

    @Override
    public Map<?, ?> getMap() {
        return map;
    }

    @Override
    public Object getRequestMessage() {
        return request;
    }

    @Override
    public Object getResponseMessage() {
        return response;
    }

    @Override
    public void setRequestMessage(final Object request) {
        this.request = (HttpServletRequest) request;
    }

    @Override
    public void setResponseMessage(final Object response) {
        this.response = (HttpServletResponse) response;
    }

}
