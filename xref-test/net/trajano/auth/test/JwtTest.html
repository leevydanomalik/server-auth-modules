30  public class JwtTest {
31      private JsonWebKeySet jwks;
32      private PrivateKey privateKey;
33  
34      @Before
35      public void setKeys() throws GeneralSecurityException {
36          final KeyPairGenerator kpg = KeyPairGenerator.getInstance("RSA");
37          kpg.initialize(2048);
38          final KeyPair kp = kpg.genKeyPair();
39          final String e = Base64.encodeWithoutPadding(((RSAPublicKey) kp
40                  .getPublic()).getPublicExponent().toByteArray());
41          final String n = Base64.encodeWithoutPadding(((RSAPublicKey) kp
42                  .getPublic()).getModulus().toByteArray());
43          privateKey = kp.getPrivate();
44          jwks = new JsonWebKeySet(createObjectBuilder().add(
45                  "keys",
46                  createArrayBuilder().add(
47                          createObjectBuilder().add("kty", "RSA")
48                          .add("alg", "RS256").add("use", "sig")
49                          .add("kid", "1234").add("e", e).add("n", n)))
50                          .build());
51      }
52  
53      @Test
54      public void testEnum() {
55          assertEquals(JsonWebAlgorithm.RS256,
56                  Enum.valueOf(JsonWebAlgorithm.class, "RS256"));
57      }
58  
59      @Test(expected = JsonParsingException.class)
60      public void testInvalidToken() throws Exception {
61          Utils.getJwsPayload("ABCD", jwks);
62      }
63  
64      @Test
65      public void testNoCryptoToken() throws Exception {
66          final String joseHeader = "{\"alg\":\"none\"}";
67          final byte[] message = "HELLO".getBytes("UTF-8");
68  
69          final byte[] jwsPayload = Utils.getJwsPayload(
70                  Base64.encodeWithoutPadding(joseHeader.getBytes("UTF-8")) + "."
71                          + Base64.encodeWithoutPadding(message), jwks);
72          assertArrayEquals(message, jwsPayload);
73      }
74  
75      @Test
76      public void testSignedToken() throws Exception {
77          final String joseHeader = "{\"kid\":\"1234\",\"alg\":\"RS256\"}";
78          final byte[] message = "HELLO".getBytes("UTF-8");
79          final Signature sig = Signature.getInstance("SHA256withRSA");
80          sig.initSign(privateKey);
81          sig.update((Base64.encodeWithoutPadding(joseHeader.getBytes("UTF-8"))
82                  + "." + Base64.encodeWithoutPadding(message)).getBytes("UTF-8"));
83          final byte[] sigbytes = sig.sign();
84  
85          final byte[] jwsPayload = Utils.getJwsPayload(
86                  Base64.encodeWithoutPadding(joseHeader.getBytes("UTF-8")) + "."
87                          + Base64.encodeWithoutPadding(message) + "."
88                          + Base64.encodeWithoutPadding(sigbytes), jwks);
89          assertArrayEquals(message, jwsPayload);
90      }
91  
92      @Test
93      public void testWithGoogleData() throws Exception {
94          final JsonWebKeySet jwks = new JsonWebKeySet(Json.createReader(
95                  Thread.currentThread().getContextClassLoader()
96                  .getResourceAsStream("googlecerts.json")).readObject());
97          final BufferedReader reader = new BufferedReader(new InputStreamReader(
98                  Thread.currentThread().getContextClassLoader()
99                  .getResourceAsStream("jwt.txt")));
100         System.out.println(new String(Utils.getJwsPayload(reader.readLine(),
101                 jwks), Charsets.UTF_8));
102     }
