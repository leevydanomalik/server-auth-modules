27      @Test
28      public void testCompressDecompress() throws Exception {
29          final String payload = "foo";
30          final ByteArrayOutputStream baos = new ByteArrayOutputStream();
31          final GZIPOutputStream os = new GZIPOutputStream(baos);
32          os.write(payload.toString().getBytes("UTF-8"));
33          os.close();
34          final String encoded = Base64.encodeWithoutPadding(baos.toByteArray());
35  
36          final GZIPInputStream is = new GZIPInputStream(
37                  new ByteArrayInputStream(Base64.decode(encoded)));
38          @SuppressWarnings("resource")
39          final String decoded = new Scanner(is).useDelimiter("\\A").next();
40          is.close();
41          assertEquals(payload, decoded);
42      }
