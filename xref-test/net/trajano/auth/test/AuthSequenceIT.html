25  public class AuthSequenceIT {
26      @Test
27      public void testRedirectToEndpoint() throws Exception {
28          final OpenIDConnectAuthModule module = new OpenIDConnectAuthModule();
29  
30          final MessagePolicy mockRequestPolicy = mock(MessagePolicy.class);
31          when(mockRequestPolicy.isMandatory()).thenReturn(true);
32          final Map<String, String> options = ImmutableMap.<String, String> builder()
33                  .put(OpenIDConnectAuthModule.ISSUER_URI_KEY, "https://accounts.google.com")
34                  .put(REDIRECTION_ENDPOINT_URI_KEY, "/app/oauth2")
35                  .put(OAuthModule.CLIENT_ID_KEY, "clientID")
36                  .put(OAuthModule.CLIENT_SECRET_KEY, "clientSecret")
37                  .build();
38          module.initialize(mockRequestPolicy, null, null, options);
39  
40          final MessageInfo messageInfo = mock(MessageInfo.class);
41  
42          final HttpServletRequest servletRequest = mock(HttpServletRequest.class);
43          when(servletRequest.getMethod()).thenReturn("GET");
44          when(servletRequest.getRequestURL()).thenReturn(new StringBuffer("https://i.trajano.net:8443/util/ejb2"));
45          when(servletRequest.getRequestURI()).thenReturn("/util/ejb2");
46          when(servletRequest.isSecure()).thenReturn(true);
47  
48          when(messageInfo.getRequestMessage()).thenReturn(servletRequest);
49  
50          final HttpServletResponse servletResponse = mock(HttpServletResponse.class);
51          when(messageInfo.getResponseMessage()).thenReturn(servletResponse);
52  
53          final Subject client = new Subject();
54          assertEquals(AuthStatus.SEND_CONTINUE, module.validateRequest(messageInfo, client, null));
55          verify(servletResponse).sendRedirect("https://accounts.google.com/o/oauth2/auth?client_id=clientID&response_type=code&scope=openid&redirect_uri=https://i.trajano.net:8443/app/oauth2&state=L3V0aWwvZWpiMg");
56      }
