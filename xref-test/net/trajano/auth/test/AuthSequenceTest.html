110     @Test
111     public void testHandleCallBack() throws Exception {
112 
113         final KeyPairGenerator kpg = KeyPairGenerator.getInstance("RSA");
114         kpg.initialize(1024);
115         final KeyPair kp = kpg.genKeyPair();
116         final String e = Base64.encodeWithoutPadding(((RSAPublicKey) kp.getPublic()).getPublicExponent()
117                 .toByteArray());
118         final String n = Base64.encodeWithoutPadding(((RSAPublicKey) kp.getPublic()).getModulus()
119                 .toByteArray());
120 
121         final JsonObject jwks = createObjectBuilder().add("keys", createArrayBuilder().add(createObjectBuilder().add("kty", "RSA")
122                 .add("alg", "RS256")
123                 .add("use", "sig")
124                 .add("kid", "1234")
125                 .add("e", e)
126                 .add("n", n)))
127                 .build();
128 
129         final OpenIDConnectAuthModule module = new OpenIDConnectAuthModule();
130 
131         final Client mockRestClient = mock(Client.class);
132         module.setRestClient(mockRestClient);
133         final WebTarget openIdConfigurationTarget = mock(WebTarget.class);
134         when(mockRestClient.target(URI.create("https://accounts.google.com/.well-known/openid-configuration"))).thenReturn(openIdConfigurationTarget);
135         final Builder openIdConfigurationBuilder = mock(Builder.class);
136         when(openIdConfigurationTarget.request(MediaType.APPLICATION_JSON_TYPE)).thenReturn(openIdConfigurationBuilder);
137 
138         when(openIdConfigurationBuilder.get(OpenIDProviderConfiguration.class)).thenReturn(googleOpenIdConfiguration);
139 
140         final WebTarget tokenEndpointTarget = mock(WebTarget.class);
141         when(mockRestClient.target(googleOpenIdConfiguration.getTokenEndpoint())).thenReturn(tokenEndpointTarget);
142         final Builder tokenEndpointBuilder = mock(Builder.class);
143         when(tokenEndpointTarget.request(MediaType.APPLICATION_JSON_TYPE)).thenReturn(tokenEndpointBuilder);
144         when(tokenEndpointBuilder.header(eq("Authorization"), anyString())).thenReturn(tokenEndpointBuilder);
145 
146         final OAuthToken oauthToken = new OAuthToken();
147 
148         final byte[] joseHeader = "{\"kid\":\"1234\",\"alg\":\"RS256\"}".getBytes(UTF_8);
149         final byte[] claimsJson = ("{\"aud\":\"clientID\",\"azp\":\"clientID\",\"exp\":" + (System.currentTimeMillis() / 1000 + 86400) + ",\"iss\":\"accounts.google.com\",\"sub\":\"12312-2312\"}").getBytes(UTF_8);
150         final Signature sig = Signature.getInstance("SHA256withRSA");
151         sig.initSign(kp.getPrivate());
152         sig.update((Base64.encodeWithoutPadding(joseHeader) + "." + Base64.encodeWithoutPadding(claimsJson)).getBytes(UTF_8));
153         final byte[] sigbytes = sig.sign();
154 
155         oauthToken.setIdToken(Base64.encodeWithoutPadding(joseHeader) + "." + Base64.encodeWithoutPadding(claimsJson) + "." + Base64.encodeWithoutPadding(sigbytes));
156 
157         when(tokenEndpointBuilder.post(any(Entity.class), eq(OAuthToken.class))).thenReturn(oauthToken);
158 
159         final WebTarget jwksTarget = mock(WebTarget.class);
160         when(mockRestClient.target(googleOpenIdConfiguration.getJwksUri())).thenReturn(jwksTarget);
161         final Builder jwksBuilder = mock(Builder.class);
162         when(jwksTarget.request(MediaType.APPLICATION_JSON_TYPE)).thenReturn(jwksBuilder);
163 
164         when(jwksBuilder.get(JsonObject.class)).thenReturn(jwks);
165 
166         final MessagePolicy mockRequestPolicy = mock(MessagePolicy.class);
167         when(mockRequestPolicy.isMandatory()).thenReturn(true);
168 
169         final CallbackHandler handler = mock(CallbackHandler.class);
170         module.initialize(mockRequestPolicy, null, handler, options);
171 
172         final MessageInfo messageInfo = mock(MessageInfo.class);
173 
174         final HttpServletRequest servletRequest = mock(HttpServletRequest.class);
175         when(servletRequest.getMethod()).thenReturn("GET");
176         when(servletRequest.getRemoteAddr()).thenReturn("8.8.8.8");
177         when(servletRequest.getRequestURL()).thenReturn(new StringBuffer("https://i.trajano.net:8443/app/oauth2?code=SplxlOBeZQQYbYS6WxSbIA&state=L3V0aWwvZWpiMg"));
178         when(servletRequest.getRequestURI()).thenReturn("/app/oauth2");
179         when(servletRequest.getContextPath()).thenReturn("/myapp");
180         when(servletRequest.isSecure()).thenReturn(true);
181         when(servletRequest.getParameter("code")).thenReturn("SplxlOBeZQQYbYS6WxSbIA");
182         when(servletRequest.getParameter("state")).thenReturn("L3V0aWwvZWpiMg");
183 
184         when(messageInfo.getRequestMessage()).thenReturn(servletRequest);
185 
186         final HttpServletResponse servletResponse = mock(HttpServletResponse.class);
187         when(servletResponse.encodeRedirectURL(anyString())).thenReturn("/util/ejb2");
188         when(messageInfo.getResponseMessage()).thenReturn(servletResponse);
189 
190         final Subject client = new Subject();
191         assertEquals(AuthStatus.SEND_SUCCESS, module.validateRequest(messageInfo, client, null));
192         verify(servletResponse).sendRedirect("/util/ejb2");
193         verify(handler).handle(any(Callback[].class));
194     }
