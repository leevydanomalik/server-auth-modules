27      @Test
28      public void testRequestMethod() {
29          final HttpServletRequest getRequest = mock(HttpServletRequest.class);
30          when(getRequest.getMethod()).thenReturn("GET");
31  
32          final HttpServletRequest headRequest = mock(HttpServletRequest.class);
33          when(headRequest.getMethod()).thenReturn("HEAD");
34  
35          final HttpServletRequest postRequest = mock(HttpServletRequest.class);
36          when(postRequest.getMethod()).thenReturn("POST");
37  
38          assertTrue(Utils.isGetRequest(getRequest));
39          assertFalse(Utils.isGetRequest(headRequest));
40          assertFalse(Utils.isGetRequest(postRequest));
41  
42          assertFalse(Utils.isHeadRequest(getRequest));
43          assertTrue(Utils.isHeadRequest(headRequest));
44          assertFalse(Utils.isHeadRequest(postRequest));
45  
46          assertTrue(Utils.isRetrievalRequest(getRequest));
47          assertTrue(Utils.isRetrievalRequest(headRequest));
48          assertFalse(Utils.isRetrievalRequest(postRequest));
49      }
