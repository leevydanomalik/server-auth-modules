19  public class TestUtilityClasses {
20      @Test
21      public void testBase64() {
22          assertArrayEquals(new byte[] {},
23                  Base64.decode(Base64.encode(new byte[] {})));
24          assertArrayEquals(new byte[] { 1 },
25                  Base64.decode(Base64.encode(new byte[] { 1 })));
26          assertArrayEquals(new byte[] { 1, 2 },
27                  Base64.decode(Base64.encode(new byte[] { 1, 2 })));
28          assertArrayEquals(new byte[] { 1, 2, 3 },
29                  Base64.decode(Base64.encode(new byte[] { 1, 2, 3 })));
30      }
31  
32      @Test
33      public void testBase64Range() {
34          final byte[] buf = new byte[] { 1, 2, 3 };
35          assertArrayEquals(new byte[] {},
36                  Base64.decode(Base64.encode(buf, 0, 0)));
37          assertArrayEquals(new byte[] { 1 },
38                  Base64.decode(Base64.encode(buf, 0, 1)));
39          assertArrayEquals(new byte[] { 1, 2 },
40                  Base64.decode(Base64.encode(buf, 0, 2)));
41          assertArrayEquals(new byte[] { 1, 2, 3 },
42                  Base64.decode(Base64.encode(buf, 0, 3)));
43      }

45      @Test
46      public void testIsNullOrEmpty() {
47          assertTrue(isNullOrEmpty(null));
48          assertTrue(isNullOrEmpty(""));
49          assertTrue(isNullOrEmpty(" "));
50          assertFalse(isNullOrEmpty("archie"));
51      }

53      @Test
54      public void testRequestMethod() {
55          final HttpServletRequest getRequest = mock(HttpServletRequest.class);
56          when(getRequest.getMethod()).thenReturn("GET");
57  
58          final HttpServletRequest headRequest = mock(HttpServletRequest.class);
59          when(headRequest.getMethod()).thenReturn("HEAD");
60  
61          final HttpServletRequest postRequest = mock(HttpServletRequest.class);
62          when(postRequest.getMethod()).thenReturn("POST");
63  
64          assertTrue(Utils.isGetRequest(getRequest));
65          assertFalse(Utils.isGetRequest(headRequest));
66          assertFalse(Utils.isGetRequest(postRequest));
67  
68          assertFalse(Utils.isHeadRequest(getRequest));
69          assertTrue(Utils.isHeadRequest(headRequest));
70          assertFalse(Utils.isHeadRequest(postRequest));
71  
72          assertTrue(Utils.isRetrievalRequest(getRequest));
73          assertTrue(Utils.isRetrievalRequest(headRequest));
74          assertFalse(Utils.isRetrievalRequest(postRequest));
75      }
